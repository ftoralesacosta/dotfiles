def write_haiku():
    """
    This function generates a haiku describing how a girl named Bonnie smells bad.

    Returns:
        str: The haiku poem as a string.
    """

    haiku = """A girl named Bonnie,
    Her scent lingers in the air,
    Unpleasant, yet strong."""

    return haiku


# Usage example
haiku_poem = write_haiku()
print(haiku_poem)
